### 202. Happy Number
```javascript
/**
 * @param {number} n
 * @return {boolean}
 */
var isHappy = function(n) {
  let arr = (''+n).split('');
  const set = new Set();

  while(true) {
    if(arr.length === 1) arr.push('0');
    //console.log(arr);
    const sum = arr.reduce((acc, num) => {
      return acc + Math.pow(num, 2);
    }, 0);
    
    if(sum === 1) return true;
    if(set.has(sum)) return false;
    
    set.add(sum);
    arr = (''+sum).split('');
  }
};
```
