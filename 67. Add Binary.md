### 67. Add Binary
```javascript
/**
 * @param {string} a
 * @param {string} b
 * @return {string}
 */
var addBinary = function(a, b) {
  return solution1(a, b)
};

// string add
// time complexity O(n) space complexity O(n)
const solution1 = (a, b) => {
  const maxLength = Math.max(a.length, b.length);
  a = a.padStart(maxLength, '0');
  b = b.padStart(maxLength, '0');
  const result = [];
  
  let add = 0;
  for(let i = a.length - 1; i >= 0; i--) {
    const sum = +a[i] + +b[i] + add;
    if(sum === 3) {
      add = 1;
      result.unshift('1');
    } else if(sum === 2) {
      add = 1;
      result.unshift('0');
      
    } else if(sum === 1) {
      add = 0;
      result.unshift('1');
    } else {
      add = 0;
      result.unshift('0');
    }
  }
  
  result.unshift(add ? '1' : '');
  
  return result.join('');
}
```
