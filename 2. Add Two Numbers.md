### 2. Add Two Numbers
```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} l1
 * @param {ListNode} l2
 * @return {ListNode}
 */
var addTwoNumbers = function(l1, l2) {
  return solution2(l1, l2);
};

const solution2 = (l1, l2) => {
  const dummy1 = new ListNode();
  dummy1.next = l1;
  l1 = dummy1;
  
  const dummy2 = new ListNode();
  dummy2.next = l2;
  l2 = dummy2;
  

  let carry = 0
  const head = new ListNode();
  let current = head;
  
  while(l1.next || l2.next) {
    let sum = 0;
    
    if(l1.next && l2.next){
      sum = l1.next.val + l2.next.val + carry;
      l1 = l1.next;
      l2 = l2.next;
    } else if(l1.next && !l2.next) {
      if(carry){
        sum = l1.next.val + carry;
        l1 = l1.next;
      } else {
        current.next = l1.next;
        return head.next;
      }

    } else {
      if(carry){
        sum = l2.next.val + carry;
        l2 = l2.next;
      } else {
        current.next = l2.next;
        return head.next;
      }
    }
    

    carry = Math.floor(sum/10);
    current.next = new ListNode(sum%10);
    current = current.next;
  }
  
  if(carry){
    current.next = new ListNode(1);
  }

  return head.next;
}
```
